# Davenport-Heintz (Result is d == 1 or d == -1)
>>> davenport_heintz = Ex(c, All((b, a),
...    Implies(Or(And(a == d, b == c), And(a == c, b == 1)), a**2 == b)))
>>> qe(davenport_heintz)
And(d != 0, Or(d - 1 == 0, d + 1 == 0))

# Kahan's Problem

>>> sum(1 for _ in qe(ellipse).atoms())
110
