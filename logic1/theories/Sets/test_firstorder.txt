>>> from logic1.theories.Sets import *
>>> a, b, x, y, z = VV.get('a', 'b', 'x', 'y', 'z')

[x, y, y]

fvars:
>>> f = C(2) & All(y, Ex(x, (x == y) & (y == a)) & Ex(z, x == a))
>>> list(f.fvars())
[a, x, a]

qvars:
>>> list(All(y, Ex(x, a == y) & Ex(z, a == y)).qvars())
[y, x, z]
