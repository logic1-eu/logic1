>>> from logic1 import *
>>> from logic1.theories.Sets import *

>>> a, b, c, d, o = VV.get('a', 'b', 'c', 'd', 'o')

>>> dnf(F)

>>> dnf(((a != o) | (b == o)) & ((a == o) | (b == o)))
